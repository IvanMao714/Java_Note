1.1.5 字符串字数统计

public void Count(){
        String s3 = "34324";
        int numberCount = 0;
        int smallCount = 0;
        int bigCount = 0;

        for (int i = 0; i < s3.length(); i++) {
            char ch = s3.charAt(i);
            if(ch >='A' && ch<='Z'){
                bigCount++;
            }else if (ch >='a' && ch<='z') {
                smallCount++;
            }else if (ch >='0' && ch<='9') {
                numberCount++;
            }
        }
    }

1.1.7 字符串的反转

public String reverse(String s){
        String ss = "";
        for (int i = 0; i < s.length()-1; i++) {
            ss += s.charAt(i);
        }
        return ss;
    }

public void transform(){

    }

1.2.4 Case1:append

public String append1(int[] arr){
        StringBuilder sb = new StringBuilder();
        sb.append("[");
        for (int i = 0; i < arr.length; i++) {
            if (i == sb.length()-1){
                sb.append(arr[i]);
            }
            else {
                sb.append(arr[i]).append(",");
            }
        }
        sb.append("]");
        String s = sb.toString();
        return s;
    }

1.2.5 Case2: reverse

public String reverse1(String s){
        StringBuilder sb = new StringBuilder(s);
        sb.reverse();
        return sb.toString();
    }
